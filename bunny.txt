#include <stdio.h>
int iterativeBinarySearch(int arr[], int n, int key)
{
    int low = 0, high = n - 1;
    while (low <= high)
    {
        int mid = (low + high) / 2;
        if (arr[mid] == key)
        {
            return mid;
        }
        else if (key < arr[mid])
        {
            high = mid - 1;
        }
        else
        {
            low = mid + 1;
        }
    }
    return -1;
}

int recursiveBinarySearch(int arr[], int low, int high, int key)
{
    if(low > high)
        return -1;

        return recursiveBinarySearch(arr, mid + 1, high, key);
}

2) Sort a given set of n integer elements using Merge Sort method and compute its
time complexity. Demonstrate this algorithm using Divide-and-Conquer method.

#include <stdio.h>
void merge(int a[], int l, int m, int h)
{
    int i, j, k;
    int n1 = m - l + 1, n2 = h - m;
    int L[n1], R[n2];

    for (i = 0; i < n1; i++)
        L[i] = a[l + i];
    for (j = 0; j < n2; j++)
        R[j] = a[m + 1 + j];
    i = 0;
    j = 0;
    k = l;
    while (i < n1 && j < n2)
        a[k++] = (L[i] <= R[j]) ? L[i++] : R[j++];
    while (i < n1)
        a[k++] = L[i++];
    while (j < n2)
        a[k++] = R[j++];
}

void mergeSort(int a[], int l, int h)
{
    if (l < h)
    {
        int m = (l + h) / 2;
        mergeSort(a, l, m);
        mergeSort(a, m + 1, h);
        merge(a, l, m, h);
    }
}

int main()
{
    int n, i;
    printf("Enter number of elements: ");
    scanf("%d", &n);

    int a[n];
    printf("Enter %d integers:\n", n);
    for (i = 0; i < n; i++)
        scanf("%d", &a[i]);
    mergeSort(a, 0, n - 1);
    printf("Sorted array:\n");
    for (i = 0; i < n; i++)
        printf("%d ", a[i]);
    return 0;
}

3) Sort a given set of n integer elements using Quick Sort method and compute its time
complexity. Demonstrate this algorithm using Divide-and-Conquer method.

#include <stdio.h>
void swap(int *a, int *b)
{
    int t = *a;
    *a = *b;
    *b = t;
}

int partition(int a[], int l, int h)
{
    int p = a[h], i = l - 1;
    for (int j = l; j < h; j++)
        if (a[j] <= p)
        {
            i++;
            swap(&a[i], &a[j]);
        }
    swap(&a[i + 1], &a[h]);
    return i + 1;
}

void quickSort(int a[], int l, int h)
{
    if (l < h)
    {
        int p = partition(a, l, h);
        quickSort(a, l, p - 1);
        quickSort(a, p + 1, h);
    }
}

int main()
{
    int n;
    printf("Enter number of elements: ");
    scanf("%d", &n);
    int a[n];
    printf("Enter %d integers:\n", n);
    for (int i = 0; i < n; i++)
        scanf("%d", &a[i]);
    quickSort(a, 0, n - 1);
    printf("Sorted array:\n");
    for (int i = 0; i < n; i++)
        printf("%d ", a[i]);
    return 0;
}

6) Implement Horspool's String matching algorithm.
#include <stdio.h>
#include <string.h>
#define MAX 256
void shiftTable(char pattern[], int table[])
{
    int m = strlen(pattern);
    for (int i = 0; i < MAX; i++)
        table[i] = m;
    for (int i = 0; i < m - 1; i++)
        table[(int)pattern[i]] = m - 1 - i;
}

int horspool(char text[], char pattern[])
{
    int table[MAX];
    shiftTable(pattern, table);
    int n = strlen(text);
    int m = strlen(pattern);
    int i = m - 1;

    while (i < n)
    {
        int k = 0;
        while (k < m && pattern[m - 1 - k] == text[i - k])
            k++;
        if (k == m)
            return i - m + 1;
        i += table[(int)text[i]];
    }
    return -1;
}

int main()
{
    char text[100], pattern[100];
    printf("Enter text: ");
    scanf("%s", text);
    printf("Enter pattern: ");
    scanf("%s", pattern);
    int pos = horspool(text, pattern);

    if (pos >= 0)
        printf("Pattern found at position: %d\n", pos);
    else
        printf("Pattern not found\n");
    return 0;
}

9) Implement 0/1 Knapsack problem using Dynamic Programming Memory Functions
technique.
#include <stdio.h>
int dp[100][100];
int knapsack(int n, int W, int profit[], int weight[])
{
    if (n == 0 || W == 0)
        return 0;

    if (dp[n][W] != -1)
        return dp[n][W];

    if (weight[n - 1] > W)
    {
        dp[n][W] = knapsack(n - 1, W, profit, weight);
    }
    else
    {
        int include = profit[n - 1] +
                      knapsack(n - 1, W - weight[n - 1], profit, weight);

        int exclude = knapsack(n - 1, W, profit, weight);

        dp[n][W] = (include > exclude) ? include : exclude;
    }
    return dp[n][W];
}

int main()
{
    int n, W;
    printf("Enter number of items: ");
    scanf("%d", &n);
    int profit[n], weight[n];

    printf("Enter profits of items:\n");
    for (int i = 0; i < n; i++)
        scanf("%d", &profit[i]);

    printf("Enter weights of items:\n");
    for (int i = 0; i < n; i++)
        scanf("%d", &weight[i]);

    printf("Enter capacity of knapsack: ");
    scanf("%d", &W);
    
    for (int i = 0; i <= n; i++)
    {
        for (int j = 0; j <= W; j++)
        {
            dp[i][j] = -1;
        }
    }
    int result = knapsack(n, W, profit, weight);
    printf("Maximum Profit = %d\n", result);
    return 0;
}

#include <stdio.h>
#define N 10
#define INF 9999
int main()
{
    int n, cost[N][N], visited[N] = {0};
    int min, a = 0, b = 0;
    int mincost = 0, ne = 1;
    printf("Enter number of vertices: ");
    scanf("%d", &n);
    printf("Enter cost matrix (use 9999 for infinity):\n");
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < n; j++)
        {
            scanf("%d", &cost[i][j]);
        }
    }
    visited[0] = 1;
    printf("Edges in MST:\n");
    while (ne < n)
    {
        min = INF;

        for (int i = 0; i < n; i++)
        {
            if (visited[i])
            {
                for (int j = 0; j < n; j++)
                {
                    if (!visited[j] && cost[i][j] < min)
                    {
                        min = cost[i][j];
                        a = i;
                        b = j;
                    }
                }
            }
        }
        printf("%d -> %d = %d\n", a, b, min);
        mincost += min;
        visited[b] = 1;
        ne++;
    }
    printf("Minimum cost = %d\n", mincost);

10b) Find Minimum Cost Spanning Tree for a given weighted graph using Kruskal's algorithm.

#include <stdio.h>
#define N 10
#define INF 9999
int parent[N];
int find(int i)
{
    while (parent[i])
        i = parent[i];
    return i;
}

int uni(int i, int j)
{
    if (i != j)
    {
        parent[j] = i;
        return 1;
    }
    return 0;
}

int main()
{
    int n, cost[N][N];
    int mincost = 0, ne = 1;
    int a, b, u, v, min;
    printf("Enter number of vertices: ");
    scanf("%d", &n);
    printf("Enter cost matrix:\n");

    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < n; j++)
        {
            scanf("%d", &cost[i][j]);

            if (cost[i][j] == 0)
                cost[i][j] = INF;
        }
    }
    printf("Edges in MST:\n");
    while (ne < n)
    {
        min = INF;
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (cost[i][j] < min)
                {
                    min = cost[i][j];
                    a = u = i;
                    b = v = j;
                }
            }
        }
        u = find(u);
        v = find(v);
        if (uni(u, v))
        {
            printf("%d -> %d = %d\n", a, b, min);
            mincost += min;
            ne++;
        }
        cost[a][b] = cost[b][a] = INF;
    }
    printf("Minimum cost = %d\n", mincost);
    return 0;
}




11) From a given vertex in a weighted connected graph, determine the Single Source
Shortest Paths using Dijkstra's algorithm
#include <stdio.h>
#define N 10
#define INF 999
int main()
{
    int n, cost[N][N], dist[N], visited[N] = {0};
    int source, min, u;
    printf("Enter number of vertices: ");
    scanf("%d", &n);
    printf("Enter cost matrix:\n");
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < n; j++)
        {
            scanf("%d", &cost[i][j]);
            if (cost[i][j] == 0)
                cost[i][j] = INF;
        }
    }
    printf("Enter source vertex: ");
    scanf("%d", &source);
    for (int i = 0; i < n; i++)
    {
        dist[i] = cost[source][i];
    }
    visited[source] = 1;
    dist[source] = 0;
    for (int i = 1; i < n; i++)
    {
        min = INF;
        for (int j = 0; j < n; j++)
        {
            if (!visited[j] && dist[j] < min)
            {
                min = dist[j];
                u = j;
            }
        }
        visited[u] = 1;
        for (int j = 0; j < n; j++)
        {
            if (!visited[j] && dist[u] + cost[u][j] < dist[j])
            {
                dist[j] = dist[u] + cost[u][j];
            }
        }
    }
    printf("\nShortest distances:\n");
    for (int i = 0; i < n; i++)
    {
        printf("%d -> %d = %d\n", source, i, dist[i]);
    }
    return 0;
}

12) Demonstrate the working of back tracking approach for solving N-Queen's problem.

#include <stdio.h>
#define MAX 10
int board[MAX][MAX];
int N;
void printBoard()
{
    printf("\nSolution:\n\n");
    for (int i = 0; i < N; i++)
    {
        for (int j = 0; j < N; j++)
        {
            printf("%d ", board[i][j]);
        }
        printf("\n");
    }
}

int isSafe(int row, int col)
{
    int i, j;
    for (i = 0; i < row; i++)
    {
        if (board[i][col] == 1)
            return 0;
    }
    for (i = row - 1, j = col - 1; i >= 0 && j >= 0; i--, j--)
    {
        if (board[i][j] == 1)
            return 0;
    }
    for (i = row - 1, j = col + 1; i >= 0 && j < N; i--, j++)
    {
        if (board[i][j] == 1)
            return 0;
    }

    return 1;
}
int solveNQueen(int row)
{
    if (row == N)
        return 1;
    for (int col = 0; col < N; col++)
    {
        if (isSafe(row, col))
        {
            board[row][col] = 1;
            if (solveNQueen(row + 1))
                return 1;
            board[row][col] = 0;
        }
    }
    return 0;
}

int main()
{
    printf("Enter number of queens: ");
    scanf("%d", &N);
    for (int i = 0; i < N; i++)
    {
        for (int j = 0; j < N; j++)
        {
            board[i][j] = 0;
        }
    }
    if (solveNQueen(0))
    {
        printBoard();
    }
    else
    {
        printf("Solution does not exist.\n");
    }
    return 0;
}
